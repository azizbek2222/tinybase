Realtime SDK Integratsiyasi

Manage real-time authentication and data in your project with this SDK.

JAVASCRIPT MODULE 
<script type="module">
  import { TinyBase } from "https://tinybase.vercel.app/TinyBase-sdk.js";

  const tBase = new TinyBase({
    apiKey: "LOADING..."
  });

  // 1. Registration (Email and Password)
  async function signup(email, password) {
    const res = await tBase.signUp(email, password);
    console.log("Registered:", res.user.uid);
  }

  // 2. Login to the system
  async function login(email, password) {
    await tBase.signIn(email, password);
    console.log("Logged in");
  }
  
  // 3. Password Recovery (Newly Added)
async function forgotPassword(email) {
  await tBase.resetPassword(email);
  alert("Parolni tiklash xati yuborildi!");
  }

  // 4. Logout (Newly Added)
  function logout() {
  tBase.logOut().then(() => console.log("Chiqildi"));
  }


  // 5. User Status and Balance Tracking
  tBase.onAuthChange((user) => {
    if (user) {
      console.log("User:", user.email);
      
      // Real-time balance tracking
      tBase.watchBalance(user.uid, (val) => {
        console.log("Current balance:", val);
      });
    } else {
      console.log("Not logged in");
    }
  });
</script>
